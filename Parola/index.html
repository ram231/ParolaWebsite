# Parola

# User Guide

# Installation


Step 1: Open your File Manager.Application name for the File Manager varies from different smartphone. Select the Parola.apk and install it. Any technical issues would require to configure the installation in the settings of the Android device.


Insert Image Here Step 2: Allow all permission and Install. Be sure to have enough space of 100MB as the application holds data and cache altogether.

# Log in

Sign in with Google

You must have a Gmail account so you don't need to put any password. We follow the privacy of the users.

Step 1: Tap on the “Sign in with Google”, if you don’t have gmail, you are being prompt to the google sign in authentication. It also requires internet connection to gain access to the application

**Step 2**: Click the Three Lines at the top-left of the application or just swipe left to the edge of the application to locate and edit your profile name. There are cases where your gmail name is not identical to your name.

# Events

In Parola, You can view all of the events tha have been created.

Step 1: Tap the event you want to attend. The event must before the date of the event.

Step 2: Tap Attend Event if you want to be scheduled for an event you are registered. You are also given a notification on the date of the event.
Step 3: Finish. Now you are ready to scan for beacons.

class HomeBodyPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return StreamBuilder<QuerySnapshot>(
        stream: Firestore.instance
            .collection('events')
            .orderBy('timeStart', descending: true)
            .snapshots(),
        builder: (BuildContext context, snapshot) {
          if (!snapshot.hasData) {
            return SpinKitThreeBounce(color: Colors.green[200]);
          } else {
            return EventListView(
              eventDocuments: snapshot.data.documents,
            );
          }
        });
  }
}
class EventListView extends StatelessWidget {
  final List<DocumentSnapshot> eventDocuments;

  const EventListView({Key key, this.eventDocuments}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: eventDocuments.length,
      itemBuilder: (context, index) {
        String eventTitle = eventDocuments[index].data['eventName'].toString();
        DateTime eventDate = eventDocuments[index].data['eventDate'];
        String eventPic = eventDocuments[index].data['eventPicURL'].toString();
        return Card(
          shape: BeveledRectangleBorder(
              borderRadius: BorderRadius.only(
            topLeft: Radius.circular(16.0),
            topRight: Radius.circular(16.0),
          )),
          elevation: 16.0,
          child: GestureDetector(
            child: ListTile(
              title: Text(eventDocuments[index].data['eventName'].toString()),
              subtitle: Text(DateFormat.yMMMd().format(eventDate)),
              leading: new EventImage(eventPic: eventPic),
              onTap: () async {
                SharedPreferences prefs = await SharedPreferences.getInstance();
                String username = prefs.getString('username');
                Navigator.of(context).pushAndRemoveUntil(
                    MaterialPageRoute(
                        builder: (context) => DescPage(
                              eventTitle: eventDocuments[index]
                                  .data['eventName']
                                  .toString(),
                              username: username,
                            )),
                    (p) => true);
              },
            ),
          ),
        );
      },
    );
  }
}

class EventImage extends StatelessWidget {
  const EventImage({
    Key key,
    @required this.eventPic,
  }) : super(key: key);

  final String eventPic;

  @override
  Widget build(BuildContext context) {
    return CircleAvatar(
      maxRadius: 32.0,
      backgroundColor: Colors.green[300],
      child: AspectRatio(
        aspectRatio: 1.0,
        child: CachedNetworkImage(
          imageUrl: eventPic,
          placeholder: CircularProgressIndicator(),
        ),
      ),
    );
  }
}
